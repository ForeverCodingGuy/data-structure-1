void LinkList<T>::push_front(T cnt)//前插元素实现
{
	//先创建一个结点，将数据放入结点的数据域中
	LinkNode<T>* tem = new LinkNode<T>(cnt);
	//将该结点与原来首元结点相连
	tem->next = head->next;
	//再将头指针与该结点相连
	head->next = tem;
}

void LinkList<T>::push_back(T cnt)//后插元素实现
{

	LinkNode<T>* tem = new LinkNode<T>(cnt);
	tem->next = nullptr;
	//先找到插入前最后一个结点（根据next是否为nullptr），从head开始
	LinkNode<T>* p = head;
	while (p)
	{
		if (p->next == nullptr)
		{
			p->next = tem;
			break;
		}
		p = p->next;
	}
}

template<class T>//返回链表长度
unsigned int LinkList<T>::length()
{
	unsigned len = 0;
	LinkNode<T>* p = head;
	while (p->next)
	{
		len++;
		p = p->next;
	}
	return len;
}


template<class T>//按结点位置删除
bool LinkList<T>::pop_sequ(T pos)
{
	LinkNode<T>* p = head;
	if (pos > 0 && pos <= length())
	{
		for (unsigned int i = 1; i <= length(); i++)
		{
			if (i == pos)
			{
				LinkNode<T>* tem = p->next;
				p->next = p->next->next;
				delete tem;
				return true;
			}
			p = p->next;
		}
	}
	return false;
}

template<class T>//销毁单链表
void LinkList<T>::delete_list()
{
	while (head->next!=nullptr)
	{
		LinkNode<T>* tem = head->next;
		head->next = tem->next;
		delete tem;
	}
}

template<class T>//销毁单链表
LinkList<T>::~LinkList()
{
	delete_list();
}

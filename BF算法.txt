#include<iostream>
using namespace std;
int index_BF(char s1[], char s2[])
{
	int len1 = strlen(s1);
	int len2 = strlen(s2);
	int i = 0;
	int j = 0;
	while (i < len1 && j < len2)
	{
		if (s1[i] == s2[j])
		{
			i++;
			j++;
		}
		else
		{
			i = i - j + 1;
			j = 0;
		}
	}
	if (s2[j]=='\0')
	{
		return i - j + 1;
	}
	return 0;
}

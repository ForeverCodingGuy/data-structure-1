#include <iostream>
#include <string>
#include <map>
using namespace std;
int main() {
    int n;
    int p = 0, t = 0;
    string str;
    cin >> n;
    for (int i = 0; i < n; i++) {
        map<char, int> tab;
        cin >> str;
        for (int j = 0; j < str.length(); j++) {
            tab[str[j]]++;
            if (str[j] == 'P') p = j;
            if (str[j] == 'T') t = j;
        }
        if (tab['P'] == 1 && tab['T'] == 1 && tab['A'] != 0 && tab.size() == 3 && t - p != 1 && p*(t-p-1) == str.length()-t-1)
            cout << "YES" << endl;
        else
            cout << "NO" << endl;
    }
    return 0;
}
